# Show root directory
ls /

# Show hidden files also
ls -la /

# Check file system disk usage, df -h = কোথায় সমস্যা হচ্ছে?
df -h   

# Check mounted partitions
mount | column -t

# du -sh = “specific folder/file এর deep dive”, du -sh = কার কারণে সমস্যা হচ্ছে?
du -sh /home/*

# Print current working directory
pwd

# Show file type
file /etc/passwd

# Tree structure (install first if needed: sudo apt install tree)
tree /
